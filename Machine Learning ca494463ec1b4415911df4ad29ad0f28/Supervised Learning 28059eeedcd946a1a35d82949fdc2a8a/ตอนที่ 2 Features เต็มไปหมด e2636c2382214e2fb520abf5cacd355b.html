import numpy as np
import time

def loop_dot(a, b): 
	"""
	implement ‡∏ï‡∏≤‡∏°‡∏™‡∏°‡∏Å‡∏≤‡∏£‡∏ó‡∏µ‡πà 1
	  a :  input vector 
	  b :  input vector ‡∏ó‡∏µ‡πà‡∏°‡∏µ dimension ‡πÄ‡∏ó‡πà‡∏≤ a
	
	Returns:
	  x : ‡∏ú‡∏•‡∏•‡∏±‡∏û‡∏ò‡πå‡∏™‡∏∏‡∏î‡∏ó‡πâ‡∏≤‡∏¢    
	"""
	x=0
	for i in range(a.shape[0]):
	    x = x + a[i] * b[i]
	return x
