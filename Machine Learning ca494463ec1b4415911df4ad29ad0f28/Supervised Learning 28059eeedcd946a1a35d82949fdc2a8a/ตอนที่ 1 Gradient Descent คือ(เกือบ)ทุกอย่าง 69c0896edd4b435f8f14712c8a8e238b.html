def compute_cost(x, y, w, b):
	"""
	Parameters
	* ‡πÉ‡∏ô implementation ‡∏ô‡∏µ‡πâ‡∏ú‡∏π‡πâ‡πÄ‡∏Ç‡∏µ‡∏¢‡∏ô‡∏Ç‡∏≠‡πÉ‡∏ä‡πâ x,y ‡πÄ‡∏õ‡πá‡∏ô numpy array
		x => data set ‡∏ó‡∏µ‡πà‡∏°‡∏µ‡∏Ç‡∏ô‡∏≤‡∏î m ‡∏ï‡∏±‡∏ß
		y => taget set ‡∏ó‡∏µ‡πà‡∏°‡∏µ‡∏Ç‡∏ô‡∏≤‡∏î m ‡∏ï‡∏±‡∏ß
		w => weight
		b => bias
	return ‡∏Ñ‡πà‡∏≤‡∏ú‡∏•‡∏•‡∏±‡∏û‡∏ò‡πå‡∏Ç‡∏≠‡∏á cost (total_cost)
	"""
	m = x.shape[0]
	cost = 0
	
	for i in range(m):
	    f_wb = w * x[i] + b
	    cost = cost + (f_wb - y[i])**2
	total_cost = 1 / (2 * m) * cost
	
	return total_cost

def compute_gradient(x, y, w, b): 
	"""
	Parameters
	* ‡πÉ‡∏ô implementation ‡∏ô‡∏µ‡πâ‡∏ú‡∏π‡πâ‡πÄ‡∏Ç‡∏µ‡∏¢‡∏ô‡∏Ç‡∏≠‡πÉ‡∏ä‡πâ x,y ‡πÄ‡∏õ‡πá‡∏ô numpy array
		x => data set ‡∏ó‡∏µ‡πà‡∏°‡∏µ‡∏Ç‡∏ô‡∏≤‡∏î m ‡∏ï‡∏±‡∏ß
		y => taget set ‡∏ó‡∏µ‡πà‡∏°‡∏µ‡∏Ç‡∏ô‡∏≤‡∏î m ‡∏ï‡∏±‡∏ß
		w => weight
		b => bias
	Returns
	  dj_dw => derivative ‡∏Ç‡∏≠‡∏á dj/dw
	  dj_db => derivative ‡∏Ç‡∏≠‡∏á dj/db
	"""
	m = x.shape[0]    
	dj_dw = 0
	dj_db = 0
	
	for i in range(m):  
	    f_wb = w * x[i] + b 
	    dj_dw_i = (f_wb - y[i]) * x[i] 
	    dj_db_i = f_wb - y[i] 
	    dj_db += dj_db_i
	    dj_dw += dj_dw_i 
	dj_dw = dj_dw / m 
	dj_db = dj_db / m 
	    
	return dj_dw, dj_db
